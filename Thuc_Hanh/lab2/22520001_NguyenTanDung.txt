***Sinh viên hoàn thành Phần III bài tập QuanLyGiaoVu từ câu 1 đến câu 5.

 - Câu 1:
select MAHV, MALOP, NGSINH, TEN from HOCVIEN;

 - Câu 2:
SELECT HV.MAHV, HV.TEN, KQT.LANTHI, KQT.DIEM
FROM HOCVIEN HV
INNER JOIN KETQUATHI KQT ON HV.MAHV = KQT.MAHV
INNER JOIN LOP L ON HV.MALOP = L.MALOP
WHERE L.TENLOP = 'K12' AND KQT.MAMH = 'CTRR'

 - Câu 3:
SELECT HV.MAHV, HV.TEN, KQ.MAMH
FROM HOCVIEN HV
INNER JOIN KETQUATHI KQ ON HV.MAHV = KQ.MAHV
WHERE KQ.LANTHI = 1 AND KQ.DIEM >= 5;

 - Câu 4:
SELECT HV.MAHV, HV.TEN
FROM HOCVIEN HV
INNER JOIN KETQUATHI KQ ON HV.MAHV = KQ.MAHV
INNER JOIN LOP L ON HV.MALOP = L.MALOP
WHERE (L.TENLOP = 'K11' AND KQ.MAHV = 'CTRR') AND (KQ.LANTHI = 1 AND KQ.DIEM < 5);

 - Câu 5:
SELECT HV.MAHV, HV.TEN
FROM HOCVIEN HV
INNER JOIN LOP L ON HV.MALOP = L.MALOP
WHERE L.TENLOP LIKE 'K%' AND HV.MAHV NOT IN (
    SELECT KQ.MAHV
    FROM KETQUATHI KQ
    WHERE KQ.MAMH = 'CTRR' AND KQ.DIEM >= 5
);
